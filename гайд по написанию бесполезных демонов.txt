class Deamon:
    def __init__(self, name, is_async=None, keys=[]):
        self.name = name
        self.is_async = is_async
        self.keys = keys
    
    def run(self, window ,keys_arr):
        self.name(window, *keys_arr)

name - функция демона

is_async - будет ли функция ассинхронно работать, пока есть окно и флаг работы демонов или выполняться раз в такт

keys - аргументы функции. Нечто очень кривое
